NAME	= cub3D
INC_DIR	= inc
LIB_DIR	= lib
SRC_DIR	= src/
OBJ_DIR	= obj/

# - - - - - - - 
LIB		= $(LIB_DIR)/libft/libft.a
LIB_L	= $(LIB_DIR)/libft/
LIB_M	= $(LIB_DIR)/mlx/

# - - - - - - - 
RM		= rm -rf
MP		= mkdir -p
CFLAGS	= -g -Wall -Wextra -Werror
MLX_FLAGS	= -L$(MLX_DIR) -lmlx -framework OpenGL -framework AppKit
#LIBC ?
CC		= clang

# - - - - - - - - - - - -
FILES	= cub3d.c init.c free.c checker.c  parser.c  print.c  list_utils.c  map_utils.c  check_map.c  \
		check_textures_and_colors.c  render.c  parser_map.c  raycast.c  input.c  utils.c  mlx_conf.c  draw draw_utils.c 

SRC		= $(addprefix $(SRC_DIR), $(FILES))
OBJ		= $(addprefix $(OBJ_DIR), $(SRC:.c=.o))
DEP		= $(addsuffix .d, $(basename $(OBJ)))

# - - - - - - - -- - - -- -
all:
	@$(MAKE) -C $(LIB_L)
	@$(MAKE) -C $(LIB_M)
	@$(MAKE) $(NAME)

$(OBJ_DIR)%.o: %.c $(MKFL)
	$(MP) $(dir $@)
	printf "                       \rCompiling $<"
	$(CC) $(CFLAGS) -MMD -I $(INC_DIR) -c $< -o $@

$(NAME):: $(OBJ) $(LIB)
	printf "\nLinking cub3d\n"
	$(CC) $(CFLAGS) $(OBJ) $(LIB) -o $(NAME)

$(NAME)::
	echo "Hello, cub3d already compiled "

-include $(DEP)

clean:
	$(MAKE) clean -C $(LIB_M) --no-print-directory
	$(MAKE) clean -C $(LIB_L) --no-print-directory
	$(RM) $(OBJ_DIR)

fclean: clean
	$(MAKE) fclean -C $(LIB_L) --no-print-directory
	$(MAKE) clean -C $(LIB_M) --no-print-directory
	$(RM) $(NAME)

re:
	$(MAKE) fclean --no-print-directory
	$(MAKE) --no-print-directory

.PHONY: all clean fclean re
